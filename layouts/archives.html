{{- /* Archives page: card-grid folder layout of all posts. Overrides theme's layouts/archives.html. Uses recursive partial "archives/card-folder.html" for rendering. */ -}} {{ define "body-class" }}template-archives{{ end }} {{ define "main" }}

{{- /* Collect all post pages, sorted by date descending */ -}} {{- $pages := where .Site.RegularPages "Section" "post" -}} {{- $pages = $pages.ByDate.Reverse -}} {{- /* Extract unique top-level directory names from file paths. .File.Dir may use backslashes on Windows, normalize to forward slashes. */ -}} {{- $topDirs := slice -}} {{- range $pages -}} {{- $rel := replace .File.Dir "\\" "/" -}} {{- $rel = strings.TrimPrefix "post/" $rel -}} {{- $rel = strings.TrimSuffix "/" $rel -}} {{- if ne $rel "" -}} {{- $first := index (strings.Split $rel "/") 0 -}} {{- if not (in $topDirs $first) -}} {{- $topDirs = $topDirs | append $first -}} {{- end -}} {{- end -}} {{- end -}}
{{- range sort $topDirs -}} {{- $dirName := . -}} {{- $dirPrefix := print $dirName "/" -}} {{- $dirPages := slice -}} {{- range $pages -}} {{- $normDir := replace .File.Dir "\\" "/" -}} {{- if strings.HasPrefix $normDir (print "post/" $dirPrefix) -}} {{- $dirPages = $dirPages | append . -}} {{- end -}} {{- end -}} {{- partial "archives/card-folder.html" (dict "name" $dirName "prefix" $dirPrefix "pages" $dirPages ) -}} {{- end -}}
