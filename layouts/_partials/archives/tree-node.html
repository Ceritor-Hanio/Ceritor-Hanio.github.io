{{- /* Recursive partial: renders one node in the archives directory tree. Parameters (dict): name - display name of this directory prefix - path relative to content/post/, with trailing slash (e.g. "实践/计算机/") pages - all post pages whose .File.Dir starts with post/ NOTE: .File.Dir may use backslashes on Windows, so we normalize with replace "\\" "/" */ -}} {{- $name := .name -}} {{- $prefix := .prefix -}} {{- $pages := .pages -}} {{- $postPrefix := print "post/" $prefix -}} {{- /* 1. Direct articles: pages whose normalized .File.Dir equals $postPrefix */ -}} {{- $direct := slice -}} {{- range $pages -}} {{- if eq (replace .File.Dir "\\" "/") $postPrefix -}} {{- $direct = $direct | append . -}} {{- end -}} {{- end -}} {{- /* 2. Subdirectory pages: pages deeper than this directory */ -}} {{- $subPages := slice -}} {{- range $pages -}} {{- $normDir := replace .File.Dir "\\" "/" -}} {{- if and (strings.HasPrefix $normDir $postPrefix) (ne $normDir $postPrefix) -}} {{- $subPages = $subPages | append . -}} {{- end -}} {{- end -}} {{- /* 3. Immediate child directory names (deduplicated) */ -}} {{- $children := slice -}} {{- range $subPages -}} {{- $rel := strings.TrimPrefix $postPrefix (replace .File.Dir "\\" "/") -}} {{- $first := index (strings.Split $rel "/") 0 -}} {{- if and (ne $first "") (not (in $children $first)) -}} {{- $children = $children | append $first -}} {{- end -}} {{- end -}} {{- /* 4. Total count */ -}} {{- $total := add (len $direct) (len $subPages) -}} {{- /* 5. Render */ -}} {{- if gt (len $children) 0 -}}
{{ partial "helper/icon" "chevron-right" }} {{ $name }} {{ $total }}
