{{- /* Recursive partial: renders one folder as a card. Parameters (dict): name - display name of this directory prefix - path relative to content/post/, with trailing slash (e.g. "实践/计算机/") pages - all post pages whose .File.Dir starts with post/ */ -}} {{- $name := .name -}} {{- $prefix := .prefix -}} {{- $pages := .pages -}} {{- $postPrefix := print "post/" $prefix -}} {{- /* 1. Direct articles: pages whose normalized .File.Dir equals $postPrefix */ -}} {{- $direct := slice -}} {{- range $pages -}} {{- if eq (replace .File.Dir "\\" "/") $postPrefix -}} {{- $direct = $direct | append . -}} {{- end -}} {{- end -}} {{- /* 2. Subdirectory pages: pages deeper than this directory */ -}} {{- $subPages := slice -}} {{- range $pages -}} {{- $normDir := replace .File.Dir "\\" "/" -}} {{- if and (strings.HasPrefix $normDir $postPrefix) (ne $normDir $postPrefix) -}} {{- $subPages = $subPages | append . -}} {{- end -}} {{- end -}} {{- /* 3. Immediate child directory names (deduplicated) */ -}} {{- $children := slice -}} {{- range $subPages -}} {{- $rel := strings.TrimPrefix $postPrefix (replace .File.Dir "\\" "/") -}} {{- $first := index (strings.Split $rel "/") 0 -}} {{- if and (ne $first "") (not (in $children $first)) -}} {{- $children = $children | append $first -}} {{- end -}} {{- end -}} {{- /* 4. Total count */ -}} {{- $total := add (len $direct) (len $subPages) -}} {{- /* 5. Render card */ -}}

{{ $name }}

{{ $total }}
{{- /* Sub-folder grid (if any children) */ -}} {{- if gt (len $children) 0 -}}
{{- range sort $children -}} {{- $childName := . -}} {{- $childPrefix := print $prefix $childName "/" -}} {{- $childPostPrefix := print "post/" $childPrefix -}} {{- $childPages := slice -}} {{- range $subPages -}} {{- if strings.HasPrefix (replace .File.Dir "\\" "/") $childPostPrefix -}} {{- $childPages = $childPages | append . -}} {{- end -}} {{- end -}} {{- partial "archives/card-folder.html" (dict "name" $childName "prefix" $childPrefix "pages" $childPages ) -}} {{- end -}}
{{- end -}} {{- /* Direct articles in this folder */ -}} {{- if gt (len $direct) 0 -}}
    {{- range $direct -}}
  • {{ .Title }} {{- .Date | time.Format (default "Jan 02, 2006" .Site.Params.dateFormat.published) -}}
    • {{- end -}}
{{- end -}}
{{- /* End card */ -}}
