## 2.1 线性表
1. 定义：$n$ 个相同类型的数据元素的序列(==动态空间的结构体数组==)
2. 特征
	1. 有第一元素和最后元素
	2. 除最后元素以外的所有元素都有唯一后继
	3. 除第一元素以外的所有元素都有唯一前驱
3. 线性表的顺序表示和实现
	1. 顺序表示——
		线性表的起始地址，称作线性表的基地址 $LOC(a_{i})=LOC(a_{1})+(i-1)\times l,l \text{为一个数据元素所占的存储量}$
	2. 顺序表示的实现

	```c group:list tab:定义
	typedef struct {
		int item_1;  // 数据项1
		int item_m;  // 数据项m
	} ElemType;  // 数据元素类型
	typedef struct {
		ElemType *elem;  // 存储空间基址
		int length;      // 当前长度
		int listsize;    // 当前分配的存储容量
	} SqList;  // 顺序表类型	 	 
		```
	```c group:list tab:初始化
	  // 假设已定义宏
	#define LIST_INIT_SIZE 100  // 初始分配量
	#define OVERFLOW -2
	
	int main() {
		SqList L;
		// 分配初始存储空间
		L.elem = (ElemType*)malloc(
			LIST_INIT_SIZE * sizeof(ElemType)
		);
		if (!L.elem) exit(OVERFLOW);  // 存储分配失败
		L.length = 0;                 // 空表长度为0
		L.listsize = LIST_INIT_SIZE;  // 初始存储容量
		return 0;
	}
			``` 
	```c group:list tab:插入
	 // 假设已定义状态码
	#define OK 1
	#define ERROR 0
	typedef int Status;
	Status ListInsert_Sq(SqList &L, int i, ElemType e) {
		// i的合法性判定
		if (i < 1 || i > L.length + 1) 
			return ERROR;
		
		// 存储空间不足处理
		if (L.length >= L.listsize) {
			// 可添加扩容代码
			return ERROR;
		}    
		// 元素右移
		for (int j = L.length - 1; j >= i - 1; --j) {
			L.elem[j + 1] = L.elem[j];
		}
		L.elem[i - 1] = e;  // 插入元素
		++L.length;         // 表长增1
		return OK;
	}
				```
	```c group:list tab:删除
	 // 假设已定义状态码
	Status ListDelete_Sq(SqList &L, int i, ElemType &e) {
		// i的合法性判定
		if (i < 1 || i > L.length) 
			return ERROR;
		e = L.elem[i - 1];  // 保存被删元素   
		// 元素左移
		for (int j = i;j <= L.length - 1; ++j) {
			L.elem[j - 1] = L.elem[j];
		}
		--L.length;  // 表长减1
		return OK;
	}
					```
	3.优缺点 
		1.  优点
			a. 节省存储空间;
			b. 对线性表中第i 个结点的操作易于实现;
			c. 容易查找一个结点的前驱和后继。
		2. 缺点
			a. ==**插入和删除操作需要移动数据**==;
			b. 建立空表时， **较难确定所需的存储空间**。 

4. 线性表的链式表示和实现
	1. 单链表
		1. 定义：
			一个线性表由若干个 结点 组成， 每个 至少含有两个域 ： 数据域( 信息域)和 和 指针域( 链域) ，由这样的结点存储的线性表称作 链表
			- 逻辑次序和物理次序不一定相同 ；元素之间的逻辑关系用指针表示；需要额外空间存储元素之间的关系。
			1. 头指针
				第一个结点的存储地址作为线性表的基地址 
			2. 头节点
				在单链表上设置一个结点，它本身不存放数据，它的 指针域（就是头指针）指向第一个元素的地址
		2. 实现
			```c group:Tlist tab:定义
			typedef struct LNode {
				ElemType data; // 数据域
				struct LNode *next; // 指针域
			} LNode, *LinkList;
```
			```c group:Tlist tab:插入方法 
			//头插法
			void ListInsert_L(ElemType e,LNode *  L){ //L 是头节点
				LNode * p= (LNode *) malloc (sizeof (LNode));
				p->data=e;
				p->next=L->next;
				L->next=p;
			}
			//尾插法
			void ListInsert_R(ElemType e,LNode *  R){ //R是尾节点
				LNode * p= (LNode *) malloc (sizeof (LNode));
				p->data=e;
				p->next=NULL;
				r->next=p;//在目前的尾部后面接上p
				r=p;//将p标记为新尾部
			}
	5.```
		```c group:Tlist tab:查找
		Status GetElem_L(LinkList L, int i, ElemType &e) {
			p = L->next; j = 1; // p 指向第一个结点，j 为计数器
			while (p && j<i) / / 顺指针向后查找, 直到 p 指向第 i 个元素或p 为空
			{
				p = p->next;
				++j;
			}
			if ( !p||j>i)// i≥1,j 的初始值是1 ，j>i 表示出现了i<1 的情况，是不合理的
				return ERROR; // 第 i 个元素不存在
			e = p->data; // 取得第 i 个元素
			return OK;
		}
		```
		```c group:Tlist tab:删除
		Status ListDelete_L(LinkList &L, int i, ElemType &e) {
			p = L; j = 0;// 用p 存储待删结点前驱
			while (p->next && j < i-1) { p = p->next; ++j; }
			// p 的后继代表要删除的元素，因此p->next 不能为空
			if (!(p->next) ||j>i-1) // i 不在1 和表长之间
				return ERROR; // 删除位置不合理
			q = p->next;
			p->next = q->next;
			e = q->data;
			free(q);
			return OK;
		} // ListDelete_L
		```
		3. 优缺点
			1. 优点
				a. 插入和删除不需要移动数据（时间复杂性低）;
				b. 不需预先分配空间 。
			2. 缺点
				a. 指针占用存储空间，增加了内存负担 。
				b. 只能顺序查找（操作）
				c. 只带头指针的单链表不容易找尾指针
	2. 循环链表
		单链表最后一个结点的指针域指向头节点
	3. 双向链表
		一个链表的每一个结点含有两个指针域：一个指针指向其前驱结点，另一个指针指向其后继结点
	```c group:Dlist tab:定义
	typedef struct DuLNode{
		ElemType data;
		struct DuLNode *prior;
		struct DuLNode *next;
	} DuLNode,*DuLinkList;
			```
	```c group:Dlist tab:插入
	Status ListInsert_Dul(DuLinkList &L, int i ,ElemType e)
	{ 
		if(!(p=GetElemP_DuL(L,i)))
			return error;

			return error;

	```
	```c group:Dlist tab:删除

	{
		if(!(p=GetElemP_DuL(L,i)))
			return error;

		p->next->prior=p->prior;
		free(p);

	}

- 定义

		}
		 ```

## 2.3 队列
一种特殊的线性表， 只允许在表的一端插入，另一端删除
- 链队列
```c group:queue tab:定义
typedef struct QNode { // 结点类型
	QElemType data;
	struct QNode *next;
} QNode, *QueuePtr;
typedef struct { // 链队列类型
	QueuePtr front; // 队头指针
	QueuePtr rear; // 队尾指针
} LinkQueue;
LinkQueue Q;
```
```c group:queue tab:初始化
Status InitQueue (LinkQueue &Q) { // 构造一个空队列Q
	Q.front = Q.rear = (QueuePtr)malloc(sizeof(QNode));
	if (!Q.front)
		exit (OVERFLOW); // 存储分配失败
	Q.front->next = NULL;
	return OK;
}
```
```c group:queue tab:入队
Status EnQueue (LinkQueue &Q, QElemType e){ // 插入元素e 为Q 的新的队尾元素
	p = (QueuePtr) malloc (sizeof (QNode));
	if (!p) exit (OVERFLOW);
	p->data = e;
	p->next = NULL;
	Q.rear->next = p;
	Q.rear = p;
	return OK;
}


```
```c group:queue tab:出队
Status DeQueue (LinkQueue &Q, QElemType &e) {
	if (Q.front == Q.rear) return ERROR;
	p = Q.front->next; e = p->data;
	Q.front->next = p->next;
	if (Q.rear == p) Q.rear = Q.front; // 删除的p 恰好是队尾即Q 只有一个元素时
	free (p);
	return OK;
}
```

### 2.4.2 模式匹配算法(KMP)
此处不讲KMP的具体过程，思想就是利用已经匹配成功的
```c group:kmp tab:next数组
void get_next(SString T,int next[]){
	int j = 1,k = 0;//j为待匹配序列位置，k为原序列位置
	next[1] = 0;
	while(j<T[0]){
		if(k == 0|| T[j]==T[k]){
			++j;++k;next[j]=k;
		}
		else k = next[j];
	}
}
```
```c group:kmp tab:匹配
int Index_KMP(SString S,SString T,int pos){//返回匹配的起始位置
	i = pos,j = 1;
	while(i <= S[0] && j <= T[0]){
		if(j == 0||S[i] == T[j]){i++;j++;}
		else j = next[j];
	}
	if(j > T[0]) return i - T[0];
	return 0;
}
